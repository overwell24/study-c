```c
#include<stdio.h>
int MemberPayment(int price, char grade)
{           
    int result;
    
    if (grade == 'S')
    {
        result = price * 0.95;
        return result;
    }
    else if (grade == 'G')
    {

        result = price * 0.90;
        return result;
    }
    else if (grade == 'V')
    {
        result = price * 0.85;
        return result;
    }

}

int main(void)
{
    int price;
    char grade;

    printf("가격 입력:");
    scanf("%d", &price);

    printf("grade 입력:");
    scanf(" %c", &grade);

    int result;
    result = MemberPayment(price,grade);
    printf("할인된 가격: %d", result);
    return 0;
}
```
