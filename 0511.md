```c
#include<stdio.h>
int MemberPayment(int price, char grade)
{           
    int result;
    
    if (grade == 'S')
    {
        result = price * 0.95;
        return result;
    }
    else if (grade == 'G')
    {

        result = price * 0.90;
        return result;
    }
    else if (grade == 'V')
    {
        result = price * 0.85;
        return result;
    }

}

int main(void)
{
    int price;
    char grade;

    printf("가격 입력:");
    scanf("%d", &price);

    printf("grade 입력:");
    scanf(" %c", &grade);

    int result;
    result = MemberPayment(price,grade);
    printf("할인된 가격: %d", result);
    return 0;
}
```

## 문제3 소수 구별
```c
#include <stdio.h>

int isPrime(int x);

int main(){
    
    int x;
    scanf("%d", &x);
    int check;
    check = isPrime(x);
    if(check){
        printf("소수입니다!");
    }
    return 0;
}

int isPrime(int x){
    
    int count=0;
    
    for(int i=1;i<=x;i++){
        
        if(x%i==0 && x%x==0){
            count++;
        }
    }
    if (count==2){
        return 1;
    }
    else{
        return 0;
    }

}
```

## 문제3 랜덤
```c
#include <stdio.h>
#include <time.h>
#define MAX 45


int main(void)
{
    srand((unsigned)time(NULL));
    for (int i = 0; i < 6; i++)
    {
        printf("%d", 1+srand()% MAX);
    }
    
    
    return 0;
}
```
